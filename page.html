üõçÔ∏è Shopping List

      
Interactive prompt to build a shopping list and show totals.

      
# shopping_list.py
shopping = {}
while True:
  item = input("Item (or done): ").strip()
  if item.lower() == 'done':
    break
  try:
    price = float(input("Price: "))
  except ValueError:
    print("Invalid price, try again")
    continue
  shopping[item] = price

print('\nYour shopping list:')
for k, v in shopping.items():
  print(f"{k}: {v}")
print('Total:', sum(shopping.values()))

    


    

      
üéØ Number Guessing

      
A simple game: guess the secret number.

      
# guess.py
import random
secret = random.randint(1, 100)
while True:
  try:
    n = int(input('Guess (1-100): '))
  except ValueError:
    print('Enter a number')
    continue
  if n < secret:
    print('Too low')
  elif n > secret:
    print('Too high')
  else:
    print('Correct!')
    break

    

    


    

    

      
üß∞ Container Manager

      
Add/remove items in a small dictionary and print it without braces.

      
# container.py
data = {'key': 'value'}
while True:
  cmd = input('add/remove/print/exit: ').strip().lower()
  if cmd == 'add':
    k = input('key: ')
    v = input('value: ')
    data[k] = v
  elif cmd == 'remove':
    k = input('key to remove: ')
    data.pop(k, None)
  elif cmd == 'print':
    print(', '.join(f"{k}: {v}" for k, v in data.items()))
  elif cmd == 'exit':
    break

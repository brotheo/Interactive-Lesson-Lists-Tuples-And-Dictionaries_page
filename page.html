üéØ Sample 1: Student grades (list)

      
# Student grades example
grades = [85, 92, 78, 96, 88]
print(f"Grades: {grades}")
print(f"Average: {sum(grades) / len(grades):.1f}")
print(f"Highest: {max(grades)}")
print(f"Lowest: {min(grades)}")

# Add new grade
grades.append(91)
print(f"After adding 91: {grades}")

# Remove lowest grade
grades.remove(min(grades))
print(f"After removing lowest: {grades}")


      
üåç Sample 2: World capitals (dictionary)

      
# World capitals example
capitals = {
    'Canada': 'Ottawa',
    'France': 'Paris',
    'Japan': 'Tokyo',
    'Australia': 'Canberra'
}

# Access values
print(f"Capital of France: {capitals['France']}")

# Safe access with get()
print(f"Capital of Germany: {capitals.get('Germany', 'Unknown')}")

# Add new country
capitals['Brazil'] = 'Bras√≠lia'

# Print all capitals
for country, capital in capitals.items():
    print(f"{country}: {capital}")


      
üìä Sample 3: RGB colors (tuple)

      
# RGB color tuples example
red = (255, 0, 0)
green = (0, 255, 0)
blue = (0, 0, 255)
purple = (128, 0, 128)

colors = [red, green, blue, purple]
color_names = ['Red', 'Green', 'Blue', 'Purple']

for i, color in enumerate(colors):
    r, g, b = color  # Tuple unpacking
    print(f"{color_names[i]}: RGB({r}, {g}, {b})")
